Posons , l'unificateur principal de et . Nous appelerons la paire des deux termes en bleu et rouge une paire critique Si deux règles génèrent une paire critique, on dit qu'elles se superposent. L'existence de paires critiques est un signe d'ambiguïté du système de réécriture.

Si ces paires critiques sont joignable, le système de réécriture est alors localement confluent.

Théorème 4 (Knuth-Bendix)   Un système de réécriture noethérien est confluent si ses paires critiques sont joignables.

let rec super l1 l2 occ =
	match occ with
	| a::b -> 
	begin
		try 
		let t = cut a l1 in
		match t with 
		| Var _ -> raise Impossible
		| _ -> let sigma = unifier ((cut a l1), l2)
				in (a, sigma)
		with Impossible -> super l1 l2 b 
	end
| [] -> raise Impossible
in super l1 l2 (occurences 0 l1)

let rec cp (l1,r1) (l2,r2) =
  let (oc, sigma) = superpose l1 l2 in
