(* fonction générée en COQ*)
  or_elim = 
 fun (a b c : Prop) (h1 : a -> c) (h2 : b -> c) (h3 : a \/ b) =>
 match h3 with
  | or_introl ha => h1 ha
  | or_intror hb => h2 hb
 end
     : forall a b c : Prop, (a -> c) -> (b -> c) -> a \/ b -> c
